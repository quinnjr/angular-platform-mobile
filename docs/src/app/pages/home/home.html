Write Angular, Run Native

app.component.ts 
@Component({
  selector: 'app-root',
  template: \`
    <mobile-view [style]="styles.container">
      <mobile-text [style]="styles.title">
        Hello, Angular Mobile!
      </mobile-text>

      <mobile-button
        [style]="styles.button"
        (press)="onPress()">
        Get Started
      </mobile-button>
    </mobile-view>
  \`,
})
export class AppComponent {
  styles = StyleSheet.create({
    container: {
      flex: 1,
      justifyContent: 'center',
      alignItems: 'center',
      backgroundColor: '#f5f5f5',
    },
    title: {
      fontSize: 24,
      fontWeight: 'bold',
      marginBottom: 20,
    },
    button: {
      backgroundColor: '#dd0031',
      padding: 16,
      borderRadius: 8,
    },
  });

  onPress() {
    console.log('Button pressed!');
  }
}
