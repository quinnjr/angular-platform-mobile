Styling

Angular Platform Mobile uses a React Native-like styling system based on Flexbox. Styles are defined as JavaScript objects and applied directly to native views.

StyleSheet

Use StyleSheet.create() to define reusable styles with type checking and optimization: 

import { StyleSheet } from '@pegasusheavy/angular-platform-mobile';

const styles = StyleSheet.create({
  container: {
    flex: 1,
    justifyContent: 'center',
    alignItems: 'center',
    backgroundColor: '#ffffff',
  },
  title: {
    fontSize: 24,
    fontWeight: 'bold',
    color: '#1a1a1a',
    marginBottom: 16,
  },
  button: {
    backgroundColor: '#dd0031',
    paddingVertical: 12,
    paddingHorizontal: 24,
    borderRadius: 8,
  },
});

Style Types

ViewStyle

Used for container components like mobile-view:

const viewStyle: ViewStyle = {
  // Flexbox
  flex: 1,
  flexDirection: 'row',      // 'row' | 'column' | 'row-reverse' | 'column-reverse'
  justifyContent: 'center',  // 'flex-start' | 'flex-end' | 'center' | 'space-between' | 'space-around'
  alignItems: 'center',      // 'flex-start' | 'flex-end' | 'center' | 'stretch' | 'baseline'
  flexWrap: 'wrap',          // 'wrap' | 'nowrap' | 'wrap-reverse'

  // Dimensions
  width: 100,
  height: 200,
  minWidth: 50,
  maxWidth: '100%',
  aspectRatio: 16 / 9,

  // Spacing
  padding: 16,
  paddingVertical: 8,
  paddingHorizontal: 12,
  margin: 8,
  marginTop: 16,

  // Background
  backgroundColor: '#ffffff',
  opacity: 0.9,

  // Border
  borderWidth: 1,
  borderColor: '#cccccc',
  borderRadius: 8,
  borderTopLeftRadius: 16,

  // Position
  position: 'absolute',      // 'relative' | 'absolute'
  top: 0,
  left: 0,
  right: 0,
  bottom: 0,
  zIndex: 10,

  // Shadow (iOS)
  shadowColor: '#000000',
  shadowOffset: { width: 0, height: 2 },
  shadowOpacity: 0.25,
  shadowRadius: 4,

  // Shadow (Android)
  elevation: 4,

  // Transforms
  transform: [
    { translateX: 10 },
    { rotate: '45deg' },
    { scale: 1.2 },
  ],
};

TextStyle

Used for text components:

const textStyle: TextStyle = {
  // Font
  fontSize: 16,
  fontWeight: 'bold',        // 'normal' | 'bold' | '100'-'900'
  fontFamily: 'Roboto',
  fontStyle: 'italic',       // 'normal' | 'italic'

  // Color
  color: '#333333',

  // Alignment
  textAlign: 'center',       // 'auto' | 'left' | 'right' | 'center' | 'justify'

  // Spacing
  lineHeight: 24,
  letterSpacing: 0.5,

  // Decoration
  textDecorationLine: 'underline',  // 'none' | 'underline' | 'line-through'
  textDecorationColor: '#dd0031',
  textDecorationStyle: 'solid',

  // Transform
  textTransform: 'uppercase',  // 'none' | 'uppercase' | 'lowercase' | 'capitalize'

  // Shadow
  textShadowColor: '#000000',
  textShadowOffset: { width: 1, height: 1 },
  textShadowRadius: 2,
};

ImageStyle

Used for image components:

const imageStyle: ImageStyle = {

Dynamic Styles

Conditional Styles

Combine styles conditionally using arrays or spread operators:

@Component({
  template: \`
    <mobile-view [style]="getContainerStyle()">
      <mobile-text [style]="[styles.text, isActive && styles.activeText]">
        Status
      </mobile-text>
    </mobile-view>
  \`,
})
export class StatusComponent {
  isActive = false;
  isError = false;

  styles = StyleSheet.create({
    container: {
      padding: 16,
      backgroundColor: '#ffffff',
    },
    containerActive: {
      backgroundColor: '#e8f5e9',
    },
    containerError: {
      backgroundColor: '#ffebee',
    },
    text: {
      fontSize: 16,
      color: '#666666',
    },
    activeText: {
      color: '#4caf50',
      fontWeight: 'bold',
    },
  });

  getContainerStyle() {
    return [
      this.styles.container,
      this.isActive && this.styles.containerActive,
      this.isError && this.styles.containerError,
    ];
  }
}

StyleService

Use the StyleService for advanced styling features:

import { StyleService } from '@pegasusheavy/angular-platform-mobile';

@Component({...})
export class MyComponent {
  private styleService = inject(StyleService);

  // Use presets
  containerStyle = {
    ...this.styleService.presets.flex1,
    ...this.styleService.presets.center,
    ...this.styleService.presets.shadow,
  };

  // Create responsive styles
  responsiveText = this.styleService.responsive(
    { fontSize: 14 },        // Base
    {
      sm: { fontSize: 16 },  // >= 320px
      md: { fontSize: 18 },  // >= 480px
      lg: { fontSize: 20 },  // >= 768px
    }
  );

  // Create themed styles
  themedStyles = this.styleService.themed(
    { container: { backgroundColor: '#ffffff' } },  // Light
    { container: { backgroundColor: '#1a1a1a' } }   // Dark
  );
}

Platform-Specific Styles

Use Platform.select() for platform-specific styling:

import { Platform, StyleSheet } from '@pegasusheavy/angular-platform-mobile';

const styles = StyleSheet.create({
  button: {
    padding: 12,
    borderRadius: Platform.select({
      ios: 10,
      android: 4,
    }),
    ...Platform.select({
      ios: {
        shadowColor: '#000',
        shadowOffset: { width: 0, height: 2 },
        shadowOpacity: 0.1,
        shadowRadius: 4,
      },
      android: {
        elevation: 4,
      },
    }),
  },
});

Flexbox Guide

The layout system is based on Flexbox with flexDirection: 'column' as the default (unlike web CSS which defaults to row).

Row Layout

{
  flexDirection: 'row',
  justifyContent: 'space-between',
}

Column Centered

{
  flexDirection: 'column',
  alignItems: 'center',
}

Flex Grow

// Child with flex: 1
{ flex: 1 }
